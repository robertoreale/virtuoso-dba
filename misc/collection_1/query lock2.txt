Select s1.username || '@' || s1.machine
  || ' ( SID=' || s1.sid || ' )  is blocking '
    || s2.username || '@' || s2.machine || ' ( SID=' || s2.sid || ' ) ' AS blocking_status
  From (select sid,id1,id2 from gv$lock where BLOCK=1) l1, gv$session s1,
       (select sid,id1,id2 from gv$lock where request > 0) l2, gv$session s2
  Where s1.sid=l1.sid and s2.sid=l2.sid
  And l1.id1 = l2.id1
  And l1.id2 = l2.id2 ;
